% ====================
% Frame: Logging Structure
% ====================
\begin{frame}[t]{Logging Structure: get\_HistoryInit.m / get\_HistoryLog.m}
    \footnotesize
    \subsec{Prediction logs}
    \items{
        \item{\texttt{history.pred.pre}: baseline prediction window}
        \item{\texttt{history.pred.opt}: optimized prediction window}
        \item{stored fields: speed, acceleration, power, current, SOC, temperature, aging, and bounds}
    }

    \subsec{Replay logs}
    \items{
        \item{\texttt{history.app}: applied optimized trajectory}
        \item{\texttt{history.base}: replay-baseline trajectory}
        \item{meta fields: previous speed, traveled distance, initial position, and battery state}
    }
\end{frame}

% ====================
% Frame: Edit Order
% ====================
\begin{frame}[t]{Edit Order When EV Outputs Change}
    \footnotesize
    \subsec{Recommended update order}
    \items{
        \item{\texttt{get\_evmodel.m}: add or modify \texttt{EVout.xxx}}
        \item{\texttt{get\_pred\_hist.m}: add the prediction-side field}
        \item{\texttt{get\_appd\_hist.m}: add the replay-side field}
        \item{\texttt{get\_plot.m}: update figures only if visualization is needed}
    }

    \subsec{Current output groups}
    \items{
        \item{power: \texttt{P\_req}, \texttt{P\_trac}, \texttt{P\_cool}}
        \item{battery: \texttt{I\_batt}, \texttt{Q\_Ah}, \texttt{SOC}, \texttt{Rin}}
        \item{thermal: \texttt{T\_batt}, \texttt{Q\_batt}, \texttt{Q\_env}, \texttt{Q\_cool}}
        \item{aging: \texttt{Ah\_ir}, \texttt{Q\_loss}, \texttt{Ah\_eff}, \texttt{AgeCost}}
    }
\end{frame}
